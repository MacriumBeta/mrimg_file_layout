13This file defines a set of cross-platform wrapper functions for file operations.
14These functions provide a higher-level, more convenient interface for performing
15common file operations such as opening a file, setting the file pointer, getting
16the current file offset, and reading data from a file. They also handle error
17checking and throw exceptions if any of the operations fail, simplifying error
18handling in the code that uses these functions.
19
20Using cross-platform file operations is important as it allows the code to be
21portable and run on different operating systems without requiring system-specific
22implementations. This increases the maintainability and scalability of the code,
23making it more robust and versatile for various use cases and environments.

27// A custom deleter for std::fstream.
28// This deleter will be used with std::shared_ptr to automatically close files when they are no longer in use.
29struct fileDeleter {
30 void operator()(std::fstream* file) const {
31 if (file) {
32 file->close();
33 delete file;
34 }
35 }
36};

67std::fstream* openFile(const std::wstring& filename) {
68 std::fstream* file = new std::fstream();
69 file->open(filename, std::ios::in | std::ios::out | std::ios::binary);
70 if (!file->is_open()) {
71 char errorBuffer[100];
72 strerror_s(errorBuffer, sizeof(errorBuffer), errno);
73 std::string errorMessage = "Could not open file: " + wideToString(filename);
74 errorMessage += ". Error: ";
75 errorMessage += errorBuffer;
76 delete file;
77 throw std::runtime_error(errorMessage);
78 }
79 return file;
80}

138void readFile(std::fstream* file, void* dataBuffer, std::streamsize bytesToRead) {
139 if (!file->is_open()) {
140 throw std::runtime_error("Failed to read from file because the file stream is not open.");
141 }
142 file->read(static_cast<char*>(dataBuffer), bytesToRead);
143 if (file->fail()) {
144 std::string errorMessage = "Failed to read from file.";
145 if (file->eof()) {
146 errorMessage += " Attempted to read past the end of the file.";
147 }
148 else if (file->bad()) {
149 errorMessage += " There was an error in the underlying I/O stream.";
150 }
151 else {
152 errorMessage += " The operation failed due to an unspecified error.";
153 }
154 throw std::runtime_error(errorMessage);
155 }
156}
