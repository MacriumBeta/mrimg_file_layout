228void outputProgress(const uint64_t totalBytes, uint64_t& bytesSoFar, std::chrono::steady_clock::time_point& lastUpdateTime) {
229 static uint64_t totalBytesSoFar = 0; // Keep track of total bytes processed so far
230
231 // Get the current time
232 auto now = std::chrono::steady_clock::now();
233
234 // Calculate the time difference in seconds
235 auto timeDiff = std::chrono::duration_cast<std::chrono::seconds>(now - lastUpdateTime).count();
236
237 // If at least 1 second has passed since the last update
238 if (timeDiff >= 1) {
239 totalBytesSoFar += bytesSoFar; // Add bytes processed in this interval to total
240
241 // Calculate the progress percentage
242 int progressPercentage = (int)((totalBytesSoFar * 100) / totalBytes);
243
244 // Define the total length of the progress bar
245 const int progressBarLength = 50;
246
247 // Calculate how many blocks of the progress bar should be filled
248 int filledLength = (int)(progressPercentage / 100.0 * progressBarLength);
249
250 // Ensure filledLength does not exceed the maximum
251 if (filledLength > progressBarLength) {
252 filledLength = progressBarLength;
253 }
254
255 // Create the progress bar string
256 std::string progressBar(filledLength, '#');
257 progressBar.append(progressBarLength - filledLength, '-');
258
259 // Calculate the transfer rate in Mbits/s or Gbits/s
260 double transferRate = (double)bytesSoFar * 8 / timeDiff / 1024 / 1024; // Mbits/s
261 std::string rateUnit = "Mb/s";
262 if (transferRate > 1024) {
263 transferRate /= 1024; // Gbits/s
264 rateUnit = "Gb/s";
265 }
266
267 // Print the progress bar, transfer rate
268 std::cout << "\rProgress: [" << progressBar << "] " << progressPercentage << "% " << formatBytes(totalBytesSoFar) << " / " << formatBytes(totalBytes) << " at " << std::fixed << std::setprecision(2) << transferRate << " " << rateUnit << " " << std::flush;
269
270 // Reset bytesSoFar and lastUpdateTime for the next interval
271 bytesSoFar = 0;
272 lastUpdateTime = now;
273 }
