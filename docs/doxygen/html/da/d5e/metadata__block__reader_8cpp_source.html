46{
47 // If the block length is 0, return nullptr
48 if (Header.BlockLength == 0) {

52 // Allocate a buffer to read the block
53 std::unique_ptr<unsigned char[]> readBuffer = std::make_unique<unsigned char[]>(Header.BlockLength);
54
55 // Read the block from the file
56 readFile(fileHandle, readBuffer.get(), Header.BlockLength);

59 // The AES variant and derived key are obtained from the backup layout.
60 // The decrypted data replaces the original data in the read buffer.

67 if (Header.Flags.Compression) {
68 // Get the size of the compressed block
69 auto compressedsize = ZSTD_findFrameCompressedSize(readBuffer.get(), Header.BlockLength);
70 // Get the size of the decompressed block
71 auto decompressedsize = ZSTD_getFrameContentSize(readBuffer.get(), Header.BlockLength);
72
73 // Allocate a buffer for the decompressed data
74 std::unique_ptr<unsigned char[]> outBuffer = std::make_unique<unsigned char[]>(decompressedsize);
75
76 // Decompress the block
77 size_t zout = ZSTD_decompress(outBuffer.get(), (size_t)decompressedsize, readBuffer.get(), compressedsize);
78

80 if (ZSTD_isError(zout)) {
81 throw std::runtime_error("Failed to decompress block.");
82 }
83
84 // update the block length to the decompressed size
85 Header.BlockLength = (uint32_t)decompressedsize;

92 throw std::runtime_error("Block hash mismatch.");
93 }
94
95 // Replace the read buffer with the decompressed data
96 readBuffer = std::move(outBuffer);
97 }
