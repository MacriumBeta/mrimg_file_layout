38void _getKeyHMACSHA256(const std::array<uint8_t, KEY_LENGTH>& key_array, std::array<uint8_t, SHA256_DIGEST_LENGTH>& hmac_array)
39{
40 // Create a new HMAC context
41 auto ctx = std::unique_ptr<EVP_MD_CTX, decltype(&::EVP_MD_CTX_free)>(EVP_MD_CTX_new(), ::EVP_MD_CTX_free);
42 if (!ctx) {
43 throw std::runtime_error("Failed to create EVP_MD_CTX");
44 }
45 // Initialize the HMAC context with the key and the SHA256 algorithm
46 const EVP_MD* md = EVP_sha256();
47 if (EVP_DigestInit_ex(ctx.get(), md, NULL) != 1) {
48 throw std::runtime_error("Failed to initialize EVP_MD_CTX");
49 }
50 // Create an EVP_PKEY structure from the key
51 auto pkey = std::unique_ptr<EVP_PKEY, decltype(&::EVP_PKEY_free)>(EVP_PKEY_new_mac_key(EVP_PKEY_HMAC, NULL, key_array.data(), (int)key_array.size()), ::EVP_PKEY_free);
52 if (!pkey) {
53 throw std::runtime_error("Failed to create EVP_PKEY");
54 }
55 // Set the HMAC key
56 if (EVP_DigestSignInit(ctx.get(), NULL, md, NULL, pkey.get()) != 1) {
57 throw std::runtime_error("Failed to set HMAC key");
58 }
59 // Compute the HMAC
60 size_t len = SHA256_DIGEST_LENGTH;
61 if (EVP_DigestSignFinal(ctx.get(), hmac_array.data(), &len) != 1) {
62 throw std::runtime_error("Failed to compute HMAC");
63 }
64 return;
65}

81std::unique_ptr<EVP_CIPHER_CTX, decltype(&::EVP_CIPHER_CTX_free)> _initAESOperation(const int aes_variant, const uint8_t* pDerivedKey, const uint8_t* pIV, EncryptionOperation Type)
82{
83 static const std::map<int, std::function<const EVP_CIPHER* ()>> cipher_map = {
84 {10, [pIV]() { return pIV != nullptr ? EVP_aes_128_cbc() : EVP_aes_128_ecb(); }},
85 {12, [pIV]() { return pIV != nullptr ? EVP_aes_192_cbc() : EVP_aes_192_ecb(); }},
86 {14, [pIV]() { return pIV != nullptr ? EVP_aes_256_cbc() : EVP_aes_256_ecb(); }}
87 };
88
89 auto cipher_type = cipher_map.at(aes_variant)();
90
91 auto ctx = std::unique_ptr<EVP_CIPHER_CTX, decltype(&::EVP_CIPHER_CTX_free)>(EVP_CIPHER_CTX_new(), ::EVP_CIPHER_CTX_free);
92 if (!ctx) {
93 throw std::runtime_error("Failed to create EVP_CIPHER_CTX");
94 }
95
96 if (Type == EncryptionOperation::Encrypt) {
97 if (1 != EVP_EncryptInit_ex(ctx.get(), cipher_type, NULL, pDerivedKey, pIV)) {
98 throw std::runtime_error("Failed to initialize AES encryption");
99 }
100 }
101 else {
102 if (1 != EVP_DecryptInit_ex(ctx.get(), cipher_type, NULL, pDerivedKey, pIV)) {
103 throw std::runtime_error("Failed to initialize AES decryption");
104 }
105 }
106 return ctx;
107}

132uint8_t* formatInitializationVectorForAES(const uint8_t* pImageID, const int32_t disk_number, const int32_t partition_number, const int32_t block_index, const int dwpbkdf2Iterations, const std::array<uint8_t, KEY_LENGTH>& DerivedKey, uint8_t* pIV)
133{
134 // Prepare the IV with a set/disk/partition/block - unique IV
135 // using the imageid, disk_number, partition_number and block_index
136 uint8_t data[AES_BLOCK_SIZE];
137 memcpy(data, pImageID, 8); // Copy the first 8 bytes from the 'imageid' array into the 'data' array
138 memcpy(data + 8, &disk_number, 2); // Copy the value of 'disk_number' into the 'data' array starting at the 9th byte (index 8). It assumes that 'disk_number' is a 16-bit integer (2 bytes)
139 memcpy(data + 10, &partition_number, 2); // Copy the value of 'partition_number' into the 'data' array starting at the 11th byte (index 10). It assumes that 'partition_number' is a 16-bit integer (2 bytes)
140 memcpy(data + 12, &block_index, 4); // Copy the value of 'block_index' into the 'data' array starting at the 13th byte (index 12). It assumes that 'block_index' is a 32-bit integer (4 bytes)
141 // Hash the derived key
142 uint8_t key_hash[SHA256_DIGEST_LENGTH];
143 SHA256(DerivedKey.data(), DerivedKey.size(), key_hash); // Compute SHA-256 hash of the derived key
144 // Create a new cipher context
145 std::unique_ptr<EVP_CIPHER_CTX, decltype(&::EVP_CIPHER_CTX_free)> ctx(EVP_CIPHER_CTX_new(), ::EVP_CIPHER_CTX_free);
146 if (!ctx) {
147 throw std::runtime_error("Failed to create EVP_CIPHER_CTX");
148 }
149 // Initialize the cipher context for AES-256 encryption in ECB mode
150 if (EVP_EncryptInit_ex(ctx.get(), EVP_aes_256_ecb(), NULL, key_hash, NULL) != 1) {
151 throw std::runtime_error("Failed to initialize EVP_CIPHER_CTX");
152 }
153 int len = 0;
154 // Encrypt the data using the initialized context and the key hash
155 if (EVP_EncryptUpdate(ctx.get(), pIV, &len, data, AES_BLOCK_SIZE) != 1) {
156 throw std::runtime_error("Failed to encrypt data");
157 }
158 // Clean up is handled by the unique_ptr
159 return pIV; // Return the pointer to the IV
160}

174void getDerivedKey(const uint8_t(&imageid)[8], const std::string& strPassword, const int iterations, std::array<uint8_t, KEY_LENGTH>& DerivedKey)
175{
176 // Hash the Image ID using SHA-256
177 unsigned char imageid_hash[SHA256_DIGEST_LENGTH];
178 if (!EVP_Digest(imageid, sizeof(imageid), imageid_hash, NULL, EVP_sha256(), NULL)) {
179 throw std::runtime_error("Failed to compute hash of Image ID");
180 }
181 // Use PKCS5_PBKDF2_HMAC to generate the derived key
182 // strPassword.c_str() is the password from which to derive the key
183 // -1 means the function will calculate the length of the password string
184 // imageid_hash is the salt value for the PBKDF2 function
185 // sizeof(imageid_hash) is the length of the salt
186 // iterations is the number of iterations for the PBKDF2 function
187 // EVP_sha256() is the hash function to use
188 // KEY_LENGTH is the length of the derived key
189 // key_array.data() is the buffer where the derived key will be stored
190 if (PKCS5_PBKDF2_HMAC(strPassword.c_str(), -1, imageid_hash, sizeof(imageid_hash), iterations, EVP_sha256(), KEY_LENGTH, DerivedKey.data()) != 1) {
191 throw std::runtime_error("Failed to generate derived key"); // Throw an error if the key generation failed
192 }
193 return;
194}
