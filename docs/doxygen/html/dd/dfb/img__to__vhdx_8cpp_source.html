7This C++ program is designed to convert a Macrium Reflect X backup file to a VHDX
8file.
9
10It performs the following operations:
11
121. Reads the backup file.
132. Parses its JSON data into a backupFile object.
143. Creates and mounts a VHDX file.
154. Restores a specified disk, or the 1st disk, to the VHDX file.
16
17The output VHDX file can then be used for data recovery.
18
19
20The program uses the following libraries:
21- nlohmann JSON library for JSON data extraction.
22 License: [MIT License](https://github.com/nlohmann/json/blob/develop/LICENSE.MIT)
23- OpenSSL library for decryption.
24 License: [Apache License 2.0](https://www.openssl.org/source/license.html)
25- Zlib library, a dependency of OpenSSL.
26 License: [Zlib License](https://zlib.net/zlib_license.html)
27- ztd compression library for decompression.
28 License: [Boost Software License](https://github.com/facebook/zstd/blob/dev/LICENSE.)
29===============================================================================
30*/

73int wmain(int argc, wchar_t* argv[]) {
74
75 // Set the console code page to UTF-8
76 SetConsoleOutputCP(CP_UTF8);
77
78 try {
79 // Handle the command line parameters and get the filename, password, outputPath, diskNumber, and describe flag
80 auto [filename, password, outputPath, diskNumber, keepDiskId, describe, jsonDump] = handleCommandLineParameters(argc, argv);
81
82 // Convert the password from wide string to UTF-8 format. This is necessary because the encryption functions
83 // used later in the program expect the password to be in UTF-8 format.
84 auto passwordInUtf8Format = convertToUtf8(password);
85
86 // Read the backup file and parse its JSON data into a backupFile object
87 auto backupFile = handleBackupFile(filename, passwordInUtf8Format);
88
89 // If the 'jsonDump' flag is set, output the JSON data from the backup file and exit the program
90 if (jsonDump) {
91 std::cout << backupFile.jsonStr << std::endl;
92 return 0;
93 }
94
95 // Print the disclaimer.
96 printDisclaimer();
97
98 // If the 'describe' flag is set, output the backup file's structure and content, then exit the program.
99 if (describe) {
100 describeFile(backupFile);
101 return 0;
102 }
103 // Prepare the VHDX file name
104 std::wstring vhdxName;
105 // Create and mount the VHDX file
106 auto vhdxManager = handleVHDXFile(filename, outputPath, vhdxName, backupFile, diskNumber);
107
108 // Print the restoring and to messages
109 std::wcout << L"Restoring:\t" << filename << L"\n";
110 std::wcout << L"To:\t\t" << vhdxName << L"\n\n";
111 // Restore to the VHDX file
112 restoreDisk(filename, passwordInUtf8Format, vhdxManager.GetDiskPath(), diskNumber, keepDiskId, outputProgress);
113
114 // Update the properties of the VHDX file. This operation refreshes the system's view of the disk,
115 // which is necessary after making changes to the disk, such as restoring a backup to it.
116 if (!keepDiskId) {
117 vhdxManager.UpdateDiskProperties(); // Only required if we're mounting the disk
118 }
119 // Print the success message and wait for the user to press any key before dismounting the VHDX file and exiting the program
120 std::cout << "\n\nRestore successful.\n";
121 if (!keepDiskId) {
122 std::cout << "The restored file system(s) can be viewed in Windows File Explorer.\n";
123 std::cout << "If not visible, please check the Windows Disk Management Console\n";
124 std::cout << "and assign drive letter(s) if necessary.\n\n";
125 std::cout << "Press any key to dismount the VHDX and exit . . .\n";
126 std::cin.get();
127 }
128 }
129 catch (const std::exception& e) {
130 // Print any exceptions that occur
131 std::string exceptionMessage = e.what();
132 std::cerr << '\n' << exceptionMessage << '\n';
133
134 // If the exception message starts with "Failed to write to file", print additional information about the possible cause of the error
135 if (exceptionMessage.rfind("Failed to write to file", 0) == 0) {
136 std::cerr << "Access to the mounted VHDX file has been denied.\n";
137 std::cerr << "This issue may be due to restrictions imposed by security software installed on your system.\n";
138 std::cerr << "Please check your security settings or contact your system administrator for further assistance.\n";
139 }
140
141 // Return 1 to indicate that an error occurred
142 return 1;
143 }
144
145 // Return 0 to indicate that the program ran successfully
146 return 0;
147
148}

164file_structs::fileLayout handleBackupFile(const std::wstring& filename, const std::string& password) {
165 // Create a backupFile object to hold the parsed JSON data from the backup file
166 file_structs::fileLayout backupFile;
167
168 // Read the backup file and parse its JSON data into the backupFile object
169 readBackupFile(filename, backupFile, password);
170
171 // If the backup file is an intermediate split file, throw a runtime error
172 if (backupFile._header.split_file) {
173 throw std::runtime_error("Backup file is an intermediate split file. Use the file name of last file in the split.");
174 }
175
176 // Return the backupFile object
177 return backupFile;
178}
