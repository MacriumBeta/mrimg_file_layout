44void setNewDiskID(file_structs::Disk::DiskLayout& disk)
45{
46 // Check if the disk format is GPT
47 bool isGPT = disk._header.disk_format == ImageEnums::DiskFormat::eGPT;
48 uint16_t bytesPerSector = disk._geometry.bytes_per_sector;
49
50 if (isGPT) {
51 // If the disk is GPT, the header is at sector 1
52 gpt_header* gptHeader = reinterpret_cast<gpt_header*>(disk.track0.data() + bytesPerSector);
53
54 // Create a new GUID to prevent a disk collision
55 GUID newDiskGuid;
56 if (CoCreateGuid(&newDiskGuid) == S_OK) {
57 // Assign the new GUID to the disk
58 memcpy(&gptHeader->disk_guid, &newDiskGuid, sizeof(newDiskGuid));
59
60 // Before calculating the new CRC32 checksum, the existing checksum in the header must be set to 0.
61 gptHeader->header_crc32 = 0;
62
63 // Calculate the CRC32 checksum for the GPT header and store it in the header.
64 // The checksum is calculated over the entire header, with the checksum field itself set to 0.
65 // GPT disks are not loaded if the checksum is incorrect.
66 gptHeader->header_crc32 = Calculate_CRC32((unsigned char*)gptHeader, gptHeader->header_size, sizeof(gpt_header));
67 }
68 }
69 else {
70 // If the disk is not GPT, it's MBR. Create a new disk ID to prevent a disk collision
71 BootRecord* bootRecord = reinterpret_cast<BootRecord*>(disk.track0.data());
72 // Create a random device
73 std::random_device rd;
74 // Create a distribution that spans the full range of uint32_t
75 std::uniform_int_distribution<uint32_t> dist;
76 // Generate a random 32-bit number
77 uint32_t randomDiskId = dist(rd);
78 // Assign the new disk ID to the boot code of the MBR
79 memcpy(&bootRecord->bootCode[440], &randomDiskId, sizeof(uint32_t));
80 }
81}

91std::unique_ptr<unsigned char[]> readBlock(file_structs::fileLayout& backupLayout, const EncryptionIVParams & ivParams, std::fstream* fileHandle, data_block& block)
92{
93 if (block.block_length == 0) {
94 return nullptr;
95 }
96 // Allocate a buffer to read the block
97 std::unique_ptr<unsigned char[]> readBuffer = std::make_unique<unsigned char[]>(block.block_length);
98
99 std::streamoff fileOffset = block.file_position;
100 setFilePointer(fileHandle, fileOffset, std::ios::beg);
101 readFile(fileHandle, readBuffer.get(), block.block_length);
102 if (backupLayout._encryption.aes_type != ImageEnums::AES::eNone) {
103 int aes_variant = backupLayout._encryption.getAESValue();
104 uint8_t iv[16];
105 formatInitializationVectorForAES(backupLayout._header.imageid_binary, ivParams.disk_number, ivParams.partition_number, ivParams.index, backupLayout._encryption.key_iterations, backupLayout._encryption.derived_key, iv);
106 decryptDataWithAESCBC(aes_variant, backupLayout._encryption.derived_key.data(), iv, readBuffer.get(), block.block_length);
107 }
108 if (backupLayout._compression.compression_level != ImageEnums::CompressionType::eNone) {
109 // Get the size of the compressed block
110 auto compressedsize = ZSTD_findFrameCompressedSize(readBuffer.get(), block.block_length);
111 // Get the size of the decompressed block
112 auto decompressedsize = ZSTD_getFrameContentSize(readBuffer.get(), block.block_length);
113 // Allocate a buffer for the decompressed data
114 std::unique_ptr<unsigned char[]> decompressedBuffer = std::make_unique<unsigned char[]>(decompressedsize);
115
116 // Decompress the block
117 auto zout = ZSTD_decompress(decompressedBuffer.get(), (size_t)decompressedsize, readBuffer.get(), compressedsize);
118
119 // If decompression failed, throw an exception
120 if (ZSTD_isError(zout)) {
121 throw std::runtime_error("Failed to decompress block.");
122 }
123
124 // Update the block length to the decompressed size
125 block.block_length = static_cast<uint32_t>(decompressedsize);
126
127 // Compute the MD5 hash of the decompressed data
128 auto computedHash = computeMD5Hash(decompressedBuffer.get(), decompressedsize);
129
130 // If the computed hash does not match the expected hash, throw an exception
131 if (memcmp(computedHash.data(), block.md5_hash, sizeof(block.md5_hash)) != 0) {
132 throw std::runtime_error("Block hash mismatch.");
133 }
134
135 // Replace the read buffer with the decompressed data
136 readBuffer = std::move(decompressedBuffer);
137 }
138 // Return the read (and possibly decompressed) data
139 return readBuffer;
140}

151uint64_t calculateTotalBytes(file_structs::fileLayout& fileLayout)
152{
153 uint64_t totalBytes = 0;
154 for (auto& disk : fileLayout.disks) {
155 for (auto& partition : disk.partitions) {
156 totalBytes += partition._file_system.reserved_sectors_byte_length;
157 for (auto& block : partition.data_blocks) {
158 if (block.block_length)
159 totalBytes += partition._header.block_size;
160 }
161 }
162 }
163 return totalBytes;
164}

196void restoreDisk(const std::wstring& filePath, const std::string& password,const std::wstring targetDisk, int diskNumber, bool keepDiskId, ProgressCallback outputProgress/*= nullptr*/)
197{
198 auto targetDiskFileStream = openFileWithGuard(targetDisk);
199 auto targetDiskHandle = targetDiskFileStream.get();
200 BackupSet backupSet;
201 {
202 file_structs::fileLayout backupLayout;
203 readBackupFile(filePath, backupLayout, password);
204
205 // Read the backup set, build the index and create a map of file handles
206 createBackupSet(backupSet, filePath, password, backupLayout._header.imageid);
207 }
208 // Get the backup layout for the first file in the backup set.
209 // This is the layout of the backup file to restore with the index built from the backup set.
210 file_structs::fileLayout& backupLayout = backupSet.getBackupFileWithFullIndex();
211
212 // Get the backup file handle from the backup set
213 std::fstream* backupFileHandle = backupSet.getFileHandle(backupLayout._header.file_number);
214
215 // Calculate the total size of the backup
216 auto totalBackupSize = calculateTotalBytes(backupLayout);
217
218 uint64_t bytesProcessed = 0;
219 std::streamoff fileOffset = 0;
220
221 // Set the file pointer to the beginning of the backup file
222 setFilePointer(backupFileHandle, fileOffset, std::ios::beg);
223
224 // Select the disk for restoration based on the provided disk number
225 // diskNumber is -1 if not supplied
226 file_structs::Disk::DiskLayout diskToRestore;
227 getDiskToRestoreFromDiskNumber(backupLayout, diskNumber, diskToRestore);
228
229 // A new disk ID prevents a disk collision. However, the disk ID should remain unchanged if the disk is bootable
230 if (!keepDiskId) {
231 setNewDiskID(diskToRestore);
232 }
233 // Write the track0 data to the target disk
234 writeFile(targetDiskHandle, diskToRestore.track0.data(), (uint32_t)diskToRestore.track0.size());
235
236 // If the disk format is MBR, restore the extended partition and logical drive boot records
237 if (diskToRestore._header.disk_format == ImageEnums::DiskFormat::eMBR) {
238 for (auto& extendedPartition : diskToRestore.extendedPartitions) {
239 fileOffset = extendedPartition.offset;
240 setFilePointer(targetDiskHandle, fileOffset, std::ios::beg);
241 writeFile(targetDiskHandle, &extendedPartition.partitionSector, sizeof(extendedPartition.partitionSector));
242 }
243 }
244
245 // Record the time when the restoration process starts
246 auto lastUpdateTime = std::chrono::steady_clock::now();
247
248 // Loop through each partition in the disk to restore
249 for (auto& partition : diskToRestore.partitions) {
250 fileOffset = partition._geometry.start + partition._geometry.boot_sector_offset;
251 setFilePointer(targetDiskHandle, fileOffset, std::ios::beg);
252
253 // write FAT32 reserved sectors to the target disk
254 if (partition._file_system.reserved_sectors_byte_length > 0) {
255 // Loop through each reserved sector block in the partition
256 auto totalBytesToWrite = partition._file_system.reserved_sectors_byte_length;
257 uint32_t bytesWritten = 0;
258 int index = 0;
259 for (auto& reservedSectorBlock : partition.reserved_sectors_blocks) {
260 // If encryption is enabled, set the Initialization Vector (IV) parameters
261 EncryptionIVParams ivParams;
262 if (backupLayout._encryption.aes_type != ImageEnums::AES::eNone) {
263 ivParams.Set(backupLayout._header, diskToRestore, partition, index++);
264 }
265 // Read the block data from the backup file
266 auto blockData = readBlock(backupLayout, ivParams, backupSet.getFileHandle(reservedSectorBlock.file_number), reservedSectorBlock);
267 if (blockData != nullptr) {
268 // Calculate the amount of data to write
269 uint32_t bytesToWrite = min(reservedSectorBlock.block_length, totalBytesToWrite - bytesWritten);
270 // Write the block data to the target disk
271 writeFile(targetDiskHandle, blockData.get(), bytesToWrite);
272 bytesWritten += bytesToWrite;
273 }
274 }
275 // Update the amount of data processed
276 bytesProcessed += partition._file_system.reserved_sectors_byte_length;
277 if (outputProgress) outputProgress(totalBackupSize, bytesProcessed, lastUpdateTime);
278 }
279 int blockIndex = 0;
280 auto lcn0Start = partition._file_system.lcn0_offset;
281 // Loop through each data block in the partition
282 int index = 0;
283 for (auto& dataBlock : partition.data_blocks) {
284 // If encryption is enabled, set the Initialization Vector (IV) parameters
285 EncryptionIVParams ivParams;
286 if (backupLayout._encryption.aes_type != ImageEnums::AES::eNone) {
287 ivParams.Set(backupLayout._header, diskToRestore, partition, index++);
288 }
289 // Read the block data from the backup file
290 auto blockData = readBlock(backupLayout, ivParams, backupSet.getFileHandle(dataBlock.file_number), dataBlock);
291 if (blockData != nullptr && dataBlock.block_length != 0) {
292 // Update the amount of data processed
293 bytesProcessed += partition._header.block_size;
294 // Calculate the file offset for the data block
295 fileOffset = lcn0Start + (static_cast<uint64_t>(partition._header.block_size) * blockIndex);
296 setFilePointer(targetDiskHandle, fileOffset, std::ios::beg);
297 // Write the block data to the target disk
298 writeFile(targetDiskHandle, blockData.get(), dataBlock.block_length);
299 // Output the progress of the restoration process
300 if (outputProgress) outputProgress(totalBackupSize, bytesProcessed, lastUpdateTime);
301 }
302 ++blockIndex;
303 }
304 }
305
306 // Subtract 10 seconds from lastUpdateTime to force an update
307 lastUpdateTime -= std::chrono::seconds(10);
308 if (outputProgress) outputProgress(totalBackupSize, bytesProcessed, lastUpdateTime);
309}
