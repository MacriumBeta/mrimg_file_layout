30std::wstring prepareVhdxFileName(const std::wstring& filename, const std::wstring& outputPath) {
31 // Initialize vhdxName with the filename
32 std::wstring vhdxName = filename;
33 // Find the position of ".mrimg" in vhdxName
34 std::wstring::size_type index = vhdxName.rfind(L".mrimg");
35
36 // If ".mrimg" is found, replace it with ".vhdx"
37 if (index != std::wstring::npos) {
38 vhdxName.replace(index, 6, L".vhdx");
39 }
40 else {
41 // If ".mrimg" is not found, find the position of ".mrbak" in vhdxName
42 index = vhdxName.rfind(L".mrbak");
43 // If ".mrbak" is found, replace it with ".vhdx"
44 if (index != std::wstring::npos) {
45 vhdxName.replace(index, 6, L".vhdx");
46 }
47 }
48
49 // If outputPath is not empty, replace the path in vhdxName with outputPath
50 if (!outputPath.empty()) {
51 // If outputPath exists, replace the path in vhdxName with outputPath
52 if (std::filesystem::exists(outputPath)) {
53 std::filesystem::path p(vhdxName);
54 vhdxName = (std::filesystem::path(outputPath) / p.filename()).wstring();
55 }
56 // If outputPath does not exist, throw a runtime error
57 else {
58 throw std::runtime_error("Provided output path does not exist.");
59 }
60 }
61
62 // If the VHDX file already exists, delete it
63 if (DeleteFile(vhdxName.c_str()) == 0 && GetLastError() != ERROR_FILE_NOT_FOUND) {
64 throw std::runtime_error("Failed to delete existing VHDX file: " + std::to_string(GetLastError()));
65 }
66
67 // Return the vhdxName
68 return vhdxName;
69}
