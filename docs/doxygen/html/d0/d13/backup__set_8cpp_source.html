122void buildIndex(BackupSet& backupSet, const BackupSetFileLayouts& backupSetFileLayouts)
123{
124 // If there is no delta index, there is no need to build an index
125 if (!backupSet.backupFileLayoutForRestoration->_header.delta_index) {
126 return;
127 }
128
129 // this is the most recent full index before the delta
130 int lastFullIndexBeforeDelta = (int)getFullIndexBeforeDelta(backupSetFileLayouts);
131 // Iterate over each disk in the backup set
132 for (size_t diskIndex = 0; diskIndex < backupSet.backupFileLayoutForRestoration->disks.size(); ++diskIndex) {
133
134 auto& diskToRestore = backupSet.backupFileLayoutForRestoration->disks[diskIndex];
135
136 // Iterate over each partition in the disk
137 for (size_t partitionIndex = 0; partitionIndex < diskToRestore.partitions.size(); ++partitionIndex) {
138
139 auto& partitionToRestore = diskToRestore.partitions[partitionIndex];
140
141 // Copy the data blocks from the last full index
142 partitionToRestore.data_blocks = backupSetFileLayouts[lastFullIndexBeforeDelta].get()->disks[diskIndex].partitions[partitionIndex].data_blocks;
143
144 // Iterate over each delta file in the backup set afer the last full index
145 for (int backupSetFileIndex = lastFullIndexBeforeDelta; backupSetFileIndex >= 0; backupSetFileIndex--) {
146
147 const auto& backupFileInSet = *(backupSetFileLayouts[backupSetFileIndex]);
148
149 // Split files contain no block indexes, they only contain the header and the actual data blocks
150 if (!backupFileInSet._header.split_file) {
151 auto& partitionToMap = backupFileInSet.disks[diskIndex].partitions[partitionIndex];
152 // Copy each delta data block index to the corresponding full data block index
153 mapDeltaToFullIndex(partitionToRestore, partitionToMap);
154 }
155 }
156 }
157 }

206void createBackupSet(BackupSet& backupSet, const std::wstring& filePath, const std::string& password, const std::string& imageId)
207{
208 // Create a BackupSetFileLayouts object to hold the file layouts for the backup set
209 BackupSetFileLayouts backupSetFileLayouts;
210
211 // Create a file layout for the image file
212 file_structs::fileLayout backupFile;
213 // Setting loadIndex to false saves memory and time.
214 // The index is not needed here
215 bool loadIndex = false;
216 // Read the backup file
217 readBackupFile(filePath, backupFile, password, loadIndex);
218
219 // Find the rest of the files in this backup set
220 std::filesystem::path path(filePath);
221 std::string extension = path.extension().string();
222 if (!extension.empty()) {
223 std::filesystem::path directory = path.parent_path();
224 // Iterate over all files in the directory
225 for (const auto& fileName : std::filesystem::directory_iterator(directory)) {
226 // If the file is a regular file and has the same extension as the backup file
227 if (fileName.is_regular_file() && fileName.path().extension() == extension) {
228 // Create a new BackupFileAndHandle and fileLayout using std::make_unique for efficient memory allocation
229 //auto backupFileLayout = std::make_unique<BackupFileAndHandle>();
230 auto setbackupFile = std::make_unique<file_structs::fileLayout>();
231 try {
232 // Setting loadIndex to false saves memory and time.
233 // The index is not needed here
234 bool loadIndex = false;
235 // Read the backup file
236 readBackupFile(fileName.path().wstring(), *setbackupFile, password, loadIndex);
237 // If the image ID, increment number, and file name match the backup file
238 if (setbackupFile->_header.imageid == imageId &&
239 setbackupFile->_header.increment_number <= backupFile._header.increment_number) {
240 loadIndex = true;
241
242 // Read the backup file again with loadIndex set to true
243 readBackupFile(fileName.path().wstring(), *setbackupFile, password, loadIndex);
244 // Add the backup file handle to the backup set
245 addBackupFileToSet(backupSet, fileName.path().wstring(), *setbackupFile, backupSetFileLayouts);
246 }
247 }
248 catch (const std::exception& e) {
249 // Invalid password or corrupt file errors do not fail the entire backup set
250 // There may be files in the folder that are not part of the backup set
251 std::cerr << "Caught exception: " << e.what() << '\n';
252 }
253 }
254 }
255 // Sort the backup set by file number in descending order
256 std::sort(backupSetFileLayouts.begin(), backupSetFileLayouts.end(),
257 [](const std::shared_ptr<file_structs::fileLayout>& a, const std::shared_ptr<file_structs::fileLayout>& b) {
258 return a->_header.file_number > b->_header.file_number;
259 }
260 );
261
262 backupSet.backupFileLayoutForRestoration = backupSetFileLayouts[0];
263 // Build the full index for the backup set. This maps delta incrementals into the block index.
264 buildIndex(backupSet, backupSetFileLayouts);
265
266 }
267 else {
268 // If the backup file does not have an extension, throw an exception
269 throw std::runtime_error("populateBackupSet - invalid file extension");
270 }
271}
