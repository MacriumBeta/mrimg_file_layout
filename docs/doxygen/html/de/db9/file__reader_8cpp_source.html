10This file defines a set of cross-platform functions for reading data from a
11Macrium Reflect X backup file. These functions include reading the header offset
12and magic data, reading a metadata block, and reading the JSON data from the
13backup file. They handle various aspects such as checking for encryption and
14compression flags, decompressing data if necessary, and computing and verifying
15the MD5 hash of the data. These functions provide a high-level interface for
16accessing the contents of a Macrium Reflect X backup file.

18Using cross-platform functions is important as it allows the code to be portable

152/*
153 * Reads a Macrium Reflect X backup file and parses its JSON data.
154 *
155 * This function opens the file, reads the header offset and magic data, checks the magic data,
156 * sets the file pointer to the header offset, reads the JSON data from the file, and parses the JSON data.
157 *
158 * @param filename The name of the file to read.
159 * @throws std::runtime_error if an error occurs during reading or parsing.
160 */
161void readBackupFile(const std::wstring& filename, file_structs::fileLayout& backupLayout,const std::string password, bool loadIndex /* = true */) {
162 // Open the file and get a handle to it
163 auto fileGuard = openFileWithGuard(filename);
164 std::fstream* fileHandle = fileGuard.get();
165
166 // Calculate the offset to the end of the file
167 std::streamoff fileOffset = calculateOffset();
168
169 // Set the file pointer to the end of the file
170 setFilePointer(fileHandle, fileOffset, std::ios::end);
171
172 uint64_t headerOffset;
173 std::vector<uint8_t> magicData(MAGIC_BYTES_VX_SIZE);
174 // Read the header offset and magic data from the file
175 readHeaderOffsetAndMagicData(fileHandle, headerOffset, magicData);
176
177 // Check the magic data
178 checkMagicData(magicData);
179
180 // Set the file pointer to the header offset
181 fileOffset = headerOffset;
182 setFilePointer(fileHandle, fileOffset, std::ios::beg);
183
184 // Read the JSON data from the file
185 std::string jsonStr;
186 readFileMetadataData(backupLayout, fileHandle, jsonStr);
187 // Parse the JSON data
188 nlohmann::json j = parseJsonData(jsonStr);
189 // Convert the parsed JSON data to a fileLayout object
190 backupLayout = j;
191 // Store the JSON string in the backupLayout object for future reference
192 backupLayout.jsonStr = jsonStr;
193
194 // Set the filename in the backupLayout object
195 backupLayout.file_name = filename;
196
197 // validate password
198 if (loadIndex && backupLayout._encryption.enable) {
199 backupLayout._header.set_imageid_binary();
200 backupLayout._encryption.convert_hmac_to_binary();
201 validatePassword(backupLayout, password);
202 }
203
204 fileOffset = backupLayout._header.index_file_position;
205 setFilePointer(fileHandle, fileOffset, std::ios::beg);
206
207 // Check if the file is not split.
208 // split files have no data blocks, so we don't need to read them
209 if (false == backupLayout._header.split_file) {
210 // Iterate over each disk in the parsed file layout
211 for (auto& disk : backupLayout.disks) {
212 // Read the disk metadata
213 readDiskMetadata(backupLayout, fileHandle, disk);
214 // Iterate over each partition in the disk
215 for (auto& partition : disk.partitions) {
216 // Skip the partition metadata as it isn't used for restore
217 skipMetadata(fileHandle);
218 int32_t blockCount;
219 // Read the block count
220 readFile(fileHandle, &blockCount, sizeof(blockCount));
221 // If block count is greater than zero
222 if (blockCount > 0) {
223 // Resize the reserved sectors blocks vector to the block count
224 partition.reserved_sectors_blocks.resize(blockCount);
225 // Read the data blocks into the reserved sectors blocks vector
226 readFile(fileHandle, partition.reserved_sectors_blocks.data(), sizeof(data_block) * blockCount);
227 }
228 // Read the block count again
229 readFile(fileHandle, &blockCount, sizeof(blockCount));
230 // If block count is greater than zero
231 if (blockCount > 0) {
232 bool isdelta = backupLayout._header.delta_index;
233 if (loadIndex) {
234 if (isdelta) {
235 // Resize the delta data blocks vector to the block count
236 partition.delta_data_blocks.resize(blockCount);
237 // Read the delta data blocks into the data blocks vector
238 readFile(fileHandle, partition.delta_data_blocks.data(), sizeof(delta_data_block) * blockCount);
239 }
240 else {
241 // Resize the data blocks vector to the block count
242 partition.data_blocks.resize(blockCount);
243 // Read the data blocks into the data blocks vector
244 readFile(fileHandle, partition.data_blocks.data(), sizeof(data_block) * blockCount);
245
246 }
247 }
248 else {
249 // Skip the data blocks
250 fileOffset = isdelta ? sizeof(delta_data_block) * blockCount : sizeof(data_block) * blockCount;
251 setFilePointer(fileHandle, fileOffset, std::ios::cur);
252 }
253 }
254 }
255 }
256 }
